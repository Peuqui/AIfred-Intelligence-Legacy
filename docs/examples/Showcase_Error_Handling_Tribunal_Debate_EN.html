    try:
        user = db.query(User).filter(User.id == user_id).first()
        if user is None:
            logger.warning(f"User {user_id} not found")
        return user
    except SQLAlchemyError as e:
        logger.error(f"DB error: {e}")
        return None
    except Exception as e:
        logger.error(f"Unexpected error: {e}")
        return None
